using namespace std;

struct node
{
	char data;
	node* next;
};

int main()
{
	string s;
	int n;
	cin>>s>>n;
	int l = s.length();
	int k;
	node *first, *last;
	first = NULL, last = NULL;
	for(k=0;k<l;k++)
	{
		node *p = new node;
		p->data = s[k];
		p->next = NULL;
		if(k == 0) first = p;
		else
		{
			if(k == l - 1) p->next = first;
			last->next = p;
		}
		last = p;
	}
	last = first;
	for(k=1;k<n;k++,last=last->next);
	cout<<last->data<<endl;
	return 0;
}
