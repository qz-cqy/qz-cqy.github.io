{
    int n;

    int a[n];
    int cnt = 0;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
        if(a[i] == 0) continue;

        for(int j=0;j<i;j++)

    }
    cout<<cnt<<endl;
    return 0;
}
